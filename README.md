    private void caixa1ActionPerformed(java.awt.event.ActionEvent evt) {                                       
        bMostraCaixa.setText("Caixa1");

        for (int i = 0; i < prioritario.size(); i++) {
            if (prioritario.get(i).getTipo().equals("P-")) {

                contp = true;

            }
        }
        if (contp == true && contr == false && contc == false) {
            textMostraSenha.setText(String.valueOf(prioritario.get(0).getTipo()) + String.valueOf(prioritario.get(0).getNumero()));
            prioritario.remove(0);
            contp = false;

        } else {
            if (controle >= 1 && contr == false) {
                for (int i = 0; i < rapido.size(); i++) {
                    if (rapido.get(i).getTipo().equals("R-")) {

                        contr = true;

                    }
                }
                if (contr == true && contp == false && contc == false) {
                    textMostraSenha.setText(String.valueOf(rapido.get(0).getTipo()) + String.valueOf(rapido.get(0).getNumero()));
                    rapido.remove(0);
                    contr = false;

                } else {
                    for (int i = 0; i < comum.size(); i++) {
                        if (comum.get(i).getTipo().equalsIgnoreCase("C-")) {

                            contc = true;

                        }
                    }
                    if (contc == true && contp == false && contr == false) {
                        textMostraSenha.setText(String.valueOf(comum.get(0).getTipo()) + String.valueOf(comum.get(0).getNumero()));
                        comum.remove(0);
                        contc = false;

                    }
                }

            }
        }
    }
